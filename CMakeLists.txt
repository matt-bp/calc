cmake_minimum_required(VERSION 3.1...3.22)

project(calc
    VERSION 1.0
    LANGUAGES CXX
)

find_package(FLEX REQUIRED)
flex_target(Scanner src/scanner.l ${CMAKE_CURRENT_BINARY_DIR}/scanner.cpp)


find_package(BISON REQUIRED)
bison_target(Parser src/parser.y ${CMAKE_CURRENT_BINARY_DIR}/parser.cpp)

set(BISON_Parser_COMPILE_FLAGS "${BISON_Parser_COMPILE_FLAGS} -v")
add_flex_bison_dependency(Scanner Parser)

add_executable(calc 
	src/main.cpp 
	${BISON_Parser_OUTPUTS} 
	${FLEX_Scanner_OUTPUTS}
	)

target_link_libraries(calc PRIVATE ${FLEX_LIBRARIES} ${BISON_LIBRARIES})

# Include parser.hpp file generated by bison.
message("Source dir" ${CMAKE_CURRENT_SOURCE_DIR})
target_include_directories(calc PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/build)
